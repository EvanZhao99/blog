# 策略模式

## js版本的策略模式
```js
let strategies = {
  'S': function(salary) {
    return salary * 4
  },
  'A': function(salary) {
    return salary * 3
  },
  'B': function(salary) {
    return salary * 2
  }
}

let calculateBonus = function(level, salary) {
  return strategies[level](salary)
}

console.log(calculateBonus('S', 1000)) // 4000
```
